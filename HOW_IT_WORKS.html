1 מה הפרויקט עושה — הרעיון הגדול

הבעיה: תלמידות עובדות בתוך Safe Exam Browser (SEB). הדפדפן חוסם:

  • Ctrl+C / Ctrl+V (העתקה והדבקה)
  • PrintScreen (צילום מסך)
  • פתיחת תוכנות מתוכו

הפתרון: תוכנת Python שרצה ב-Background — לא בתוך הדפדפן, אלא כחלון Windows נפרד שה-SEB לא רואה. היא:

  • קוראת טקסט ישירות מהחלון הפעיל דרך Windows UIAutomation API
  • מזהה תגיות שהתלמידה כותבת בעורך הקוד
  • שולחת שאלות ל-Gemini AI ומציגה תשובה בחלון צף
  • מצלמת מסך ומעתיקה ל-Clipboard — הכל דרך Windows, לא דרך הדפדפן

6 reader.py — קריאת טקסט מ-SEB

זה הפתרון הטכני המרכזי. Windows UIAutomation הוא API נגישות (accessibility) של מערכת ההפעלה — נועד לקוראי מסך. SEB לא חוסם אותו כי הוא חלק מ-Windows עצמו, לא מהדפדפן.

reader.py — הפונקציה הראשית
def read_focused_text() -> str | None:
    import uiautomation as auto

    # ── שלב 0: בדוק שהחלון הפעיל לא שלנו ──────────────
    fg = auto.GetForegroundControl()    # החלון שבפוקוס עכשיו
    if fg and _should_ignore(fg.Name):
        return None    # Terminal / Notepad שלנו — תתעלם

    # ── שלב 1: קריאה מהירה מהאלמנט בפוקוס ─────────────
    ctrl = auto.GetFocusedControl()     # האלמנט שהסמן עליו
    if ctrl:
        v = _extract(ctrl)
        if v and "@@" in v:
            return v    # מצאנו! הכי מהיר

        # ── שלב 2: עלייה בהורים ─────────────────────────────
        # Monaco editor שומר טקסט בהורה, לא בתיבת הקלט עצמה
        parent = ctrl.GetParentControl()
        for _ in range(5):    # עד 5 רמות למעלה
            if parent is None: break
            v = _extract(parent)
            if v and "@@" in v: return v
            parent = parent.GetParentControl()

    # ── שלב 3: חיפוש בעץ המלא ──────────────────────────
    if fg:
        return _search_tree(fg, depth=0)    # עובר על כל הילדים עד עומק 7
reader.py — _extract() ו-_should_ignore()
# 3 דרכים לקרוא טקסט מאלמנט — מנסה עד שאחת עובדת
def _extract(ctrl) -> str | None:
    for getter in [
        lambda c: c.GetValuePattern().Value,              # שדה טקסט (input)
        lambda c: c.GetTextPattern().DocumentRange.GetText(-1),  # עורך טקסט
        lambda c: c.Name,                                   # שם האלמנט
    ]:
        try:
            v = getter(ctrl)
            if v and v.strip(): return v
        except: pass
    return None

# רשימת חלונות שמתעלמים מהם
_IGNORE_TITLES = ("seb agent", "cmd.exe", "command prompt", "powershell",
                  "windows terminal", "visual studio code",
                  "notepad++", "notepad", "wordpad",
                  ".txt", ".py", "tags_guide", "project_summary")

def _should_ignore(title: str) -> bool:
    t = (title or "").lower()
    return any(x in t for x in _IGNORE_TITLES)
למה להתעלם מ-Notepad? אם TAGS_GUIDE.txt פתוח ב-Notepad, הוא מלא @@תגיות@@ והסוכן יתבלבל. לכן כל קובץ .txt וכל Notepad מתעלמים.

למה לעלות בהורים? Monaco editor (של Ultracode) בנוי מ-divs בתוך divs. הפוקוס עלול להיות על div פנימי ריק. הפתרון: עוברים עד 5 הורים למעלה עד שמוצאים את הטקסט.

7 detector.py — זיהוי תגיות

detector.py — regex + peek_tag()
# מוצא @@TAGNAME: תוכן@@ (הנקודותיים אופציונלי, re.DOTALL = כולל שורות חדשות)
_TAG_RE = re.compile(r'@@([A-Za-z]+)\s*:?\s*(.*?)@@', re.DOTALL)

# תגיות ללא תוכן: @@COPYALL@@ @@SCREENSHOT@@ וכד'
_TAG_BARE = re.compile(r'@@(COPYALL|COPYABOVE|SCREENSHOT|SOLVEALL)@@')

KNOWN_TAGS = {"ASK", "SOLVE", "FIX", "COPY", "COPYALL",
              "COPYABOVE", "SCREENSHOT", "SOLVEALL", "ASKALL"}

def peek_tag(text: str) -> tuple[str, str] | None:
    """מוצא את התגית האחרונה בטקסט. לא מעדכן state."""
    candidates = []
    for m in _TAG_RE.finditer(text):
        tag = m.group(1).upper().strip()
        content = m.group(2).strip()
        if tag in KNOWN_TAGS:
            candidates.append((m.start(), tag, content))
    # max לפי start = התגית הכי קרובה לסוף הטקסט (האחרונה)
    if not candidates: return None
    _, tag_type, content = max(candidates, key=lambda x: x[0])
    return tag_type, content
detector.py — מניעת כפילויות עם MD5 hash
_processed_hash: str | None = None   # זוכר תגית אחרונה שעובדה

def _compute_hash(tag_type: str, content: str) -> str:
    return hashlib.md5(f"{tag_type}:{content}".encode()).hexdigest()
    # "ASK:שאלה" → מחרוזת 32 תווים קבועה. קל להשוות.

def is_new(tag_type, content) -> bool:
    return _compute_hash(tag_type, content) != _processed_hash
    # True = תגית שעוד לא עובדה

def mark_processed(tag_type, content):
    global _processed_hash
    _processed_hash = _compute_hash(tag_type, content)
    # נקרא רק אחרי שהתגית נשלחה בפועל — לא בזמן הזיהוי
למה Hash ולא השוואת מחרוזות? Hash מייצר תמיד 32 תווים, קל לאחסן ולהשוות. אם @@ASK: שאלה@@ כבר טופל — כל עוד הטקסט לא השתנה, Hash זהה ולא נשלח שוב. כשהתלמידה כותבת שאלה חדשה, Hash שונה → is_new() מחזיר True.

מנגנון ה-Debounce

הבעיה: תלמידה מקלידה @@ASK: למ — עוד לא סיימה. אם נשלח לAI על כל הקשה נבזבז טוקנים ונקבל תשובות שגויות. הפתרון: מחכים שהתגית תהיה יציבה 1.5 שניות.

main.py — לולאת ה-debounce המלאה
DEBOUNCE_WITH_CONTENT = 1.5   # @@ASK: שאלה@@ — מחכים שתסיים
DEBOUNCE_BARE         = 0.5   # @@SCREENSHOT@@ — אין מה להקליד

_pending_key   = None   # "TAG:content" שמחכה
_pending_since = None   # מתי הופיע לראשונה
_pending_data  = None   # הנתונים לשלוח ל-handler

while True:
    text = read_focused_text()   # קרא טקסט מהחלון הפעיל

    if text and "@@" in text:
        result = peek_tag(text)
        if result:
            tag_type, content = result
            key = f"{tag_type}:{content}"
            REPEATABLE = {"SCREENSHOT", "COPYALL", "COPYABOVE"}

            if not is_new(tag_type, content) and tag_type not in REPEATABLE:
                pass   # כבר עובד וה-hash תואם — דלג

            elif key != _pending_key:
                # תגית חדשה/שונה — אפס טיימר
                _pending_key   = key
                _pending_since = time.time()
                _pending_data  = (tag_type, content, text)

            else:
                # אותה תגית — האם יציבה מספיק?
                debounce = DEBOUNCE_BARE if not content else DEBOUNCE_WITH_CONTENT
                if time.time() - _pending_since >= debounce:
                    mark_processed(tag_type, content)
                    _pending_key = None
                    threading.Thread(
                        target=_handle, args=_pending_data, daemon=True
                    ).start()   # Thread: לא חוסם את הלולאה

    # אם הטקסט ריק (שינוי חלון) — לא מאפסים! ייתכן שחזרה מיד
    time.sleep(0.3)   # בדיקה כל 0.3 שניות

_handle() — מטפל בכל תגית

main.py — _handle() מחלק לפי סוג
def _handle(tag_type: str, content: str, full_text: str):

    # SCREENSHOT ── צלם מסך ושמור PNG ─────────────────
    if tag_type == "SCREENSHOT":
        img = pyautogui.screenshot()     # OS-level, לא נחסם ע"י SEB
        filename = f"screenshot_{datetime.now().strftime('%Y%m%d_%H%M%S')}.png"
        img.save(str(Path(__file__).parent / filename))
        popup_window.show("SCREENSHOT", f"✅ נשמר: {filename}")
        return

    # ASKALL ── שאלה על הקוד שמעל ─────────────────────
    if tag_type == "ASKALL":
        tag_pos = full_text.upper().find("@@ASKALL")
        code = full_text[:tag_pos].strip()   # כל הקוד שמעל התגית
        if len(code) > 2000: code = code[-2000:]   # חיסכון בטוקנים
        response = ask_ai("ASKALL", f"Code:\n{code}\n\nQuestion: {content}")
        popup_window.show("ASK", response)
        return

    # SOLVEALL ── פיתרון מלא לפי כל השדה ──────────────
    if tag_type == "SOLVEALL":
        clean = re.sub(r'@@SOLVEALL@@', '', full_text).strip()
        if len(clean) > 3000: clean = clean[:3000]
        response = ask_ai("SOLVEALL", clean)
        popup_window.show("SOLVE", response)
        return

    # COPYALL ── העתק כל השדה ל-Clipboard ──────────────
    if tag_type == "COPYALL":
        clean = re.sub(r'@@COPYALL@@', '', full_text).strip()
        pyperclip.copy(clean)    # Python קובע מה ב-Clipboard, לא הדפדפן
        popup_window.show("COPY", "✅ הועתק")
        return

    # COPYABOVE ── העתק מה שמעל התגית ─────────────────
    if tag_type == "COPYABOVE":
        tag_pos = full_text.upper().find("@@COPYABOVE@@")
        above = full_text[:tag_pos].strip()
        pyperclip.copy(above)
        return

    # COPY ── העתק טקסט ספציפי ────────────────────────
    if tag_type == "COPY":
        pyperclip.copy(content)
        return

    # FIX ── העבר קוד שמעל כ-context ─────────────────
    context = ""
    if tag_type == "FIX":
        tag_pos = full_text.upper().find("@@FIX")
        context = full_text[:tag_pos].rstrip()
        if len(context) > 1500: context = context[-1500:]

    # ASK / SOLVE / FIX → AI ─────────────────────────
    response = ask_ai(tag_type, content, context)

    mode = Settings.get("response_mode", "auto")
    if mode == "type":
        typer.append_below_tag(response)   # כתיבה ישירה לעורך
    else:
        popup_window.show(tag_type, response)   # חלון קופץ

_SYSTEM = {
    "ASK":  "You are a programming tutor. Answer in Hebrew. 2-4 sentences. No full code.",
    "SOLVE": "Output ONLY the code. No prose, no markdown fences.",
    "FIX":   "Output ONLY the corrected code. No prose.",
    "ASKALL": "Answer in Hebrew about the code provided. 3-5 sentences.",
    "SOLVEALL": "Output ONLY the complete solution. No prose.",

    "ASK": 250,      ← הסבר קצר
    "SOLVE": 600,    ← קוד בינוני
    "FIX": 600,      ← קוד מתוקן
    "ASKALL": 300,   ← תשובה מפורטת
    "SOLVEALL": 800, ← קוד ארוך
}
claude_client.py — ask_ai() עם retry
def ask_ai(tag_type: str, content: str, context: str = "") -> str:
    system  = _SYSTEM.get(tag_type, _SYSTEM["ASK"])
    max_tok = _MAX_TOKENS.get(tag_type, 300)

    # FIX: שולחים קוד + הוראה ביחד
    if tag_type == "FIX" and context:
        user_msg = f"Code:\n{context}\n\nFix instruction: {content}"
    else:
        user_msg = content

    for attempt in range(3):   # עד 3 ניסיונות
        try:
            response = _get_client().models.generate_content(
                model=Settings.get("model", "gemini-2.0-flash-lite"),
                contents=user_msg,
                config=types.GenerateContentConfig(
                    system_instruction=system,
                    max_output_tokens=max_tok,
                    temperature=0.3,   ← פחות "יצירתיות", יותר עקבי
                ),
            )
            return response.text.strip()

        except Exception as e:
            if "429" in str(e) and attempt < 2:
                time.sleep(15)   ← Rate limit: מחכה 15 שניות ומנסה שוב
            else:
                raise

def make_type_btn():
    btn = tk.Button(text="⌨️  כתוב בעורך (3...)")

    def countdown():
        for i in (2, 1):
            time.sleep(1)
            btn.config(text=f"⌨️  כתוב בעורך ({i}...)")   ← מעדכן כותרת
        time.sleep(1)
        root.destroy()     ← סוגר חלון → הפוקוס חוזר לעורך
        time.sleep(0.4)   ← 400ms — מחכה שהעורך יקבל פוקוס
        typer.type_text(content)   ← מקליד!

    btn.config(command=lambda: threading.Thread(target=countdown).start())

_lock = threading.Lock()
_active_root = None

def show(tag_type, content):
    def _run():
        global _active_root
        with _lock:
            if _active_root:
                _active_root.destroy()   ← סוגר חלון קודם לפני פתיחת חדש
            root = _build_window(tag_type, content)
            _active_root = root
        root.mainloop()   ← רץ עד שסוגרים
    threading.Thread(target=_run, daemon=True).start()

def type_text(text: str):
    delay = Settings.get("type_speed", 0.015)   ← שניות בין תו לתו

    # ניסיון 1: keyboard library — תומך Unicode ועברית
    try:
        import keyboard
        keyboard.write(text, delay=delay)
        return
    except: pass

    # Fallback: pyautogui — ASCII בלבד
    try:
        import pyautogui
        pyautogui.write(text, interval=delay)
    except Exception as e:
        print(f"[typer] שגיאה: {e}")
